```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        int n = t.size();
        vector<int> ans(n,0),vis(110,INT_MAX);
        for(int i=n-1;i>=0;i--){
            int index = INT_MAX;
            for(int j = t[i]+1;j<=100;j++){
                index = min(index,vis[j]);
            }
            if(index != INT_MAX) ans[i] = index - i;
            vis[t[i]] = i;
        }
        return ans;
    }
};
```
