# 选中的参考答案

## 1. 官方题解

```C++
class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        int count = 0;
        int m = flowerbed.size();
        int prev = -1;
        for (int i = 0; i < m; ++i) {
            if (flowerbed[i] == 1) {
                if (prev < 0) count += i / 2;
                else count += (i - prev - 2) / 2;
                prev = i;
            }
        }
        if (prev < 0) count += (m + 1) / 2;
        else count += (m - prev - 1) / 2;
        return count >= n;
    }
};
```

## 2. 空间最优

```C++
class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        for (int i = 0; i < flowerbed.size(); i += 2) {
            if (flowerbed[i] == 0) {
                if (i == flowerbed.size() - 1 || flowerbed[i + 1] == 0)
                    n--;
                else i++;
            }
        }
        return n <= 0;
    }
};
```
