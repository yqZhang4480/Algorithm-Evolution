```c++
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        // 我们从小到大，先把小的饼干分配给胃口小的同学，再分配大的
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int index = 0;
        int result = 0;
        // 先遍历胃口，再遍历小孩
        for (int i = 0; i < g.size(); i++){
            while (index < s.size() && s[index] < g[i]){
                index++;
            }
            if (index == s.size()) return result;
            else{
                result++;
                index++;
            }
        }
        return result;
    }
};
```
