```C++
class Solution {
public:
	int findMinArrowShots(vector<vector<int>>& v) {
		ranges::sort(v, [](auto&& a, auto&& b) { return a[1] < b[1]; });
		return ranges::count_if(v, [now = optional<int>{}](auto&& p) mutable {
			if (now && p[0] <= now && now <= p[1]) { return false; }
			now = p[1];
			return true;
		});
	}
};
```
