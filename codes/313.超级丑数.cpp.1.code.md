class Solution {
public:
    int nthSuperUglyNumber(int n, vector<int>& primes) {
		priority_queue<uint64_t, vector<uint64_t>, greater<uint64_t>> pq;
		pq.push(1);
		uint64_t ans = 1;
		while (n--) {
			ans = pq.top();
			pq.pop();
			for (int k : primes) {
				pq.emplace(k * ans);
				if (ans % k == 0) { break; }
			}
		}
		return ans;
	}
};
