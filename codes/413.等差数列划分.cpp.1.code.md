class Solution {
public:
	int numberOfArithmeticSlices(vector<int>& nums) {
		int n = nums.size();
		vector<int> dp(n, 0);
		for (int i = 2; i < n; i++) {
			int j = i;
			for (j; j - 2 >= 0 ; j--) {
				if (nums[j] - nums[j - 1] != nums[j - 1] - nums[j - 2]) break;
			}
			dp[i] = i - j;
		}
		return accumulate(dp.cbegin(), dp.cend(), 0);
	}
};
