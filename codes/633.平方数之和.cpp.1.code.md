class Solution {
public:
	bool judgeSquareSum(int c) {
		long long a = 0;
		long long b = std::sqrt(c);
		while (a * a + b * b != c && a <= b) {
			if (a * a + b * b > c) { b--; }
			else { a++; }
		}
		return a * a + b * b == c;
	}
};
// @lc code=end
