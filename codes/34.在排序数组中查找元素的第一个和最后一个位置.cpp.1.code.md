class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        auto p = equal_range(nums.cbegin(), nums.cend(), target);
        if (p.first == p.second) { return {-1, -1}; }
        return {int(p.first - nums.cbegin()), int(p.second - nums.cbegin() - 1)};
    }
};
