class Solution {
public:
	bool checkPossibility(vector<int>& nums) {
		size_t n = nums.size();
		nums.push_back(9999999);
		int pprev = -9999999;
		int prev = -9999999;
		int cnt = 0;
		for (int i = 0; i < n; i++) {
			if (prev > nums[i]) {
				++cnt;
				if (prev > nums[i + 1] && pprev > nums[i]) {
					return false;
				}
			}
			pprev = prev;
			prev = nums[i];
		}
		return cnt <= 1;
	}
};
// @lc code=end
