class MinStack {
public:
	std::multiset<int> m;
	std::vector<int> data;
	MinStack() {}
	void push(int val) {
		m.emplace(val);
		data.emplace_back(std::move(val));
	}
	void pop() {
		auto i = data.back();
		data.pop_back();
		m.erase(m.lower_bound(i));
	}
	int top() {
		return data.back();
	}
	int getMin() {
		return *(m.rbegin());
	}
};
