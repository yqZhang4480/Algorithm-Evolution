class Solution {
public:
	int findLHS(vector<int>& nums) {
		unordered_map<int, size_t> cnt;
		for (auto n : nums) { ++cnt[n]; }
		size_t ret = 0;
		for (auto p : cnt) {
            if (cnt.contains(p.first + 1)) {
                ret = max(p.second + cnt[p.first + 1], ret);
            }
		}
		return ret;
	}
};
