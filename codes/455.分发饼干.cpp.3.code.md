class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
		sort(g.begin(), g.end());
		sort(s.begin(), s.end());
		int cnt = 0;
		auto sit = s.cbegin();
		auto git = g.cbegin();
		while (true) {
			if (git == g.cend() || sit == s.end()) return cnt;
			if (*git <= *sit) { git++; sit++; cnt++; }
			else { sit++; } 
		}
    }
};
// @lc code=end
