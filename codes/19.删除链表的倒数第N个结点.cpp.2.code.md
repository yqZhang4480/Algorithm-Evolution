class Solution {
public:
	ListNode* removeNthFromEnd(ListNode* head, int n) {
		ListNode* prev = head;
		while (n--) { head = head->next; }
		if (!head) { return prev->ne; }
		while (head->next != nullptr) {
			prev = prev->next;
			head = head->next;
		}
		prev->next = prev->next->next;
		return head;
	}
};
