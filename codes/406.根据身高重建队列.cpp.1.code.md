class Solution {
public:
    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
		sort(people.begin(), people.end(), [&](const auto& a, const auto& b) {
			return a[0] < b[0] || (a[0] == b[0] && a[1] < b[1]);
		});
		int h = people[0][0];
		while (h < people.back()[0]) {
			bool flag = false;
			auto hi = find_if_not(people.cbegin(), people.cend(), [&](const auto& a) {
				return a[0] == h;
			});
			auto it = hi;
			if (it != people.cbegin()) {
				for (; --it != people.cbegin();) {
					if ((*it)[1] <= (*hi)[1]) {
						flag = true;
						break;
					}
				}
				if ((*it)[1] <= (*hi)[1]) {
					flag = true;
					break;
				}
				people.insert(it, *hi);
				people.erase(hi);
			}
			if (flag) h++;
		}
		return people;
    }
};
// @lc code=end
