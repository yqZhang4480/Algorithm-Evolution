class Solution {
public:
	int maxProfit(vector<int>& prices) {
		int ret = 0;
		if (prices.empty()) { return ret; }
		for (auto it = next(prices.cbegin()); it != prices.cend(); it++) {
			int p = *it - *prev(it);
			if (p > 0) { ret += p; }
		}
		return ret;
	}
};
