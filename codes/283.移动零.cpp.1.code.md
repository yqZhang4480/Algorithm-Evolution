class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        size_t first = 0;
        size_t curr = 0;
        while (curr < nums.size()) {
			if (nums[curr] != 0) {
				swap(nums[curr], nums[first++]);
			}
			++curr;
        }
    }
};
