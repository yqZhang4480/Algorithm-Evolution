class Solution {
public:
	void sortColors(vector<int>& nums) {
		int p = 1;
		auto p_begin = nums.begin();
		auto p_end = nums.begin();
		auto k = nums.begin();
		while (k != nums.end()) {
			if (*k < p) {
				swap(*(p_begin++), *k);
				if (p_begin <= p_end) { swap(*p_end, *k); }
				++p_end;
			}
			else if (*k == p) { swap(*(p_end++), *k); }
			else /*(*k > p) */ { ; }
			++k;
		}
	}
};
