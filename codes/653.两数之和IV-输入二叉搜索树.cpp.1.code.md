class Solution {
public:
	bool findNumber(TreeNode* root, int p, bool is_itself = false) {
		if (!root) { return false; }
		if (root->val == p) { return !is_itself; }
		return root->val < p ? findNumber(root->right, p) : findNumber(root->left, p);
	}
	bool findTarget(TreeNode* root, int k, TreeNode* tRoot = nullptr) {
		if (!root) { return false; }
		return findNumber(tRoot ? tRoot : root, k - root->val, true) || findTarget(root->left, k, tRoot ? tRoot : root) || findTarget(root->right, k, tRoot ? tRoot : root);
	}
};
