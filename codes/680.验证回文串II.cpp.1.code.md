class Solution {
public:
	bool vpnf(string s, string::const_iterator l, string::const_iterator r) {
		while (l <= r) {
			if (*l == *r) { l++; r--; }
			else { return false; }
		}
		return true;
	}
	bool validPalindrome(string s) {
		auto l = s.cbegin();
		auto r = s.cend() - 1;
		while (l <= r) {
			if (*l == *r) { l++; r--; }
			else {
				return vpnf(s, l + 1, r) || vpnf(s, l, r - 1); 
			}
		}
		return true;
	}
};
