class Solution {
public:
    int findMinArrowShots(vector<vector<int>>& points) {
		sort(begin(points), end(points), [&](const auto& a, const auto& b){
			return a[1] < b[1];
		});
		int cnt = 0;
		for (auto it = cbegin(points); it != cend(points);) {
			auto p = (*it)[1];
			++cnt;
			while (it != cend(points) && (*it)[0] <= p)
			{
				++it;
			}
		}
		return cnt;
    }
};
// @lc code=end
