class Solution {
public:
	vector<size_t> mfs;
	size_t merge(size_t dest, size_t index) {
		return mfs[index] = find(dest);
	}
	size_t find(size_t index) {
		if (mfs[index] == size_t(-1)) { return index; }
		return find(mfs[index]);
	}
	vector<size_t> count(size_t index) {
		auto ret = vector<size_t>(mfs.size(), 0);
		for (size_t k = 0; k < mfs.size(); ++k) {
			++ret[find(mfs[k])];
		}
		return ret;
	}
	int maxAreaOfIsland(vector<vector<int>>& grid) {
		auto size_i = grid.size();
		auto size_j = grid[0].size();
		mfs.clear();
		mfs.resize(size_i * size_j, size_t(-1));
		for (size_t k = 0; k < size_i * size_j; ++k) {
			auto i = k / size_j;
			auto j = k % size_j;
			if (grid[i][j] == 1) {
				if (i - 1 < size_i && grid[i - 1][j] == 1) { merge(k, k - size_j); }
				if (i + 1 < size_i && grid[i + 1][j] == 1) { merge(k, k + size_j); }
				if (j - 1 < size_i && grid[i][j - 1] == 1) { merge(k, k - 1); }
				if (j + 1 < size_i && grid[i][j + 1] == 1) { merge(k, k + 1); }
			}
		}
		
		return 0;
	}
};
