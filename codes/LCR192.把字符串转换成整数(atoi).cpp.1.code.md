		auto it = find_first_not_of(str, ' ');
		erase(s.begin(), it);
		int nag = s[0] == '-';
		long long ans = 0;
		for (size_t i = 0 + nag; i < s.size(); ++i) {
			ans += s[i] - '0';
			if (nag) {
				if (ans > 2147483648) {
					return -2147483648;
				}
			} else {
				if (ans > 2147483647) {
					return 2147483647;
				}
			}
			ans *= 10;
		}
		
		ans /= 10;
		return nag ? -ans : ans;
	}
};
