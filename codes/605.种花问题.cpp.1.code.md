class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        for (auto it = cbegin(flowerbed); it != cend(flowerbed); it++) {
            if ((it == cbegin(flowerbed) || *prev(it) == 0) &&
                *it == 0 &&
                (next(it) == cend(flowerbed) || *next(it) == 0)) {
                it++;
                n--;
                if (it == cend(flowerbed)) break;
            }
        }
        return n <= 0;
    }
};
