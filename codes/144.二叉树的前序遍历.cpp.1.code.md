class Solution {
	struct stack_frame {
		vector<int> ans;
		TreeNode* root;
	};
public:
	vector<int> preorderTraversal(TreeNode* root) {
		stack<stack_frame> s;
		s.emplace({}, root);
		while (!s.empty()) {
			auto frame = move(s.top());
			s.pop();
			s.emplace({})
		}
		ans.push_back(root->val);
		auto&& res1 = preorderTraversal(root->left);
		auto&& res2 = preorderTraversal(root->right);
		ans.insert(ans.end(), res1.begin(), res1.end());
		ans.insert(ans.end(), res2.begin(), res2.end());
		return ans;
	}
};
