class Solution {
public:
	using dp_element = short;
	int numSquares(const int n) {
		auto dp = vector<dp_element>(n + 1);
		for (size_t i = 1; i < dp.size(); ++i) {
			dp_element m = i;
			for (size_t j = 1; j * j <= i; ++j) {
				m = min(dp[i], static_cast<dp_element>(1 + dp[i - j * j]));
			}
			dp[i] = m;
		}
		return dp[n];
	}
};
