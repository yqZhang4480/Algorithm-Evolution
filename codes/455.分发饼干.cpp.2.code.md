class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(begin(g), end(g));
        sort(begin(s), end(s));
        auto git = cbegin(g);
        auto sit = cbegin(s);
        for (; git != cend(g) && sit != cend(s);) {
            if (*git <= *sit) {
                 git++;
            }
            sit++;
        }
        return distance(cbegin(g), git);
    }
};
