class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(begin(g), end(g));
        sort(begin(s), end(s));
        auto child = cbegin(g);
        auto cookie = cbegin(s);
        unsigned int cnt = 0;
        while (child != cend(g) && cookie != cend(s)) {
            if (*child <= *cookie) {
                child++;
                cnt++;
            }
            cookie++;
        }
        return cnt;
    }
};
