class Solution {
public:
	tuple<int, size_t> get_max(auto it1, int k) {
		auto it = max_element(it1, it1 + k);
		return { *it, it - it1 };
	}
	vector<int> maxSlidingWindow(vector<int>& nums, int k) {
		size_t max_pos;
		int max_num;
		vector<int> ret;
		tie(max_num, max_pos) = get_max(nums.begin(), k);
		for (int i = nums.size() - 1; i >= k - 1; --i) {
			if (i < max_pos) { tie(max_num, max_pos) = get_max(nums.begin() + i + 1 - k, k); }
			if (!(max_num > nums[i])) { tie(max_num, max_pos) = tuple{ nums[i], i }; }
			ret.push_back(max_num);
		}
		return ret;
	}
};
